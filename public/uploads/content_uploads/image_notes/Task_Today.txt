Function to fetch data with Model:

   public function fetch(){
        return ModelName::all();
     }
    
   OR

    public function fetch(){
        return ModelName::find($id); // if we want to fetch a single record
     }	


Function to update data with Model:
     
     public function update(Request $req){
     $update = ModelName::find($req->id);
     $update->email = $req->first_name;
     $update->password = $req->last_name;
     $update->save();
     }
